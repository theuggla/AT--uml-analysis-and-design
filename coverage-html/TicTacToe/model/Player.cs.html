using System;
using System.Linq;
using System.Collections.Generic;
 
namespace TicTacToe.Model
{
public class Player
{
private ISquareToPlayOnStrategy strategy;
 
public Player (ISquareToPlayOnStrategy squareToPlayStrategy)
{
this.strategy = squareToPlayStrategy;
}
 
public virtual Square GetSquareToPlayOn(Board board)
{
return this.strategy.GetSquareToPlayOn(board);
}
}
}
